data <- read.csv("https://opendata.ecdc.europa.eu/covid19/nationalcasedeath_eueea_daily_ei/csv", na.strings = "", fileEncoding = "UTF-8-BOM")
# Filter the country of interest that will be plotted
filtered_data <- data %>% filter(countriesAndTerritories == "Austria",  #Anders == params$country
                                 year == 2022, #Anders == params$year
                                 month == 1) #Anders == params$month

plot_ly(filtered_data,
        x = ~day,
        y = ~cases,
        type = 'bar',
        text = ~paste("Day: ", day, "<br>Cases: ", cases),
        hoverInfo = 'text') %>% 
  layout(title = paste("COVID-19 Cases in Austria in maand 1 van 2022"), #Anders "COVID-19 Cases in", params$country, "in maand", params$month, "van", params$year
         xaxis = list(title = "Day"),
         yaxis = list(title = "Number of Cases"))
plot_ly(filtered_data,
        x = ~day,
        y = ~deaths,
        type = 'bar',
        text = ~paste("Day: ", day, "<br>Deaths: ", deaths),
        hoverInfo = 'text') %>% 
  layout(title = paste("COVID-19 Deaths in Austria in maand 1 van 2022"), #Anders "COVID-19 Deaths in", params$country, "in maand", params$month, "van", params$year
