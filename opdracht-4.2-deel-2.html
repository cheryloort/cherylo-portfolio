Experiment 1

Raw cloze probability:

dat = read.csv(file="data_and_materials/SPRT_LogLin_216.csv",head=T)

lin_gam=gam(SUM_3RT_trimmed ~s(c_cloze) + s(SUB,bs="re")+
              s(ITEM,bs= "re"),
             data=dat, method = "REML")
#summary(lin_gam)
plot.gam(lin_gam, select = 1)

#y_pred <- predict(lin_gam, data.frame(x = x_new))

Log-transformed cloze probability:

log_gam=gam(SUM_3RT_trimmed ~ s(c_log_cloze) + s(SUB,bs="re")+
              s(ITEM,bs= "re"),
            data=dat,
            method = "REML")
plot.gam(log_gam, select = 1)
